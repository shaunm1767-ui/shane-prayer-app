<script>

const status = document.getElementById("status")

function log(msg,color="lime"){
const line=document.createElement("div")
line.style.color=color
line.textContent=msg
status.appendChild(line)
}

function checkImages(){

document.querySelectorAll("img").forEach(img=>{

img.onload=()=>log("✔ Image loaded: "+img.src)

img.onerror=()=>log("✖ Image FAILED: "+img.src,"red")

})

}

function checkAudio(){

const audio=document.getElementById("meditationAudio")

audio.addEventListener("canplaythrough",()=>{
log("✔ Audio ready to play")
})

audio.addEventListener("error",()=>{
log("✖ Audio failed to load","red")
})

}

checkImages()
checkAudio()
